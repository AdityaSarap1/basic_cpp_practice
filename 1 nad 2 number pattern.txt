//1 nad 2 number pattern
#include <iostream>
using namespace std;
int main() 
{
    int i, j, rows;
    cin>>rows;
    for(i=1; i<=rows; i++)
    {
        for(j=1; j<=rows; j++)
        {
            if((i+j)%2==0)
            {
                cout<<1;
            }
            else
            {
                cout<<2;
            }
        }
        cout<<endl;
    }
    return 0;
}
