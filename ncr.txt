#include<iostream>
using namespace std;

int fact(int n)
{
    int initial=1;
    for(int i=2; i<=n; i++)
    {
        initial=initial*i;
        
    }
    return initial;
}

int ncr(int n, int r, int nr)
{
    int NCR = n/(nr*r);
    return NCR;
}

int main()
{
    cout<<"Enter value of n: "<<endl;
    int n, r;
    cin>>n;
    cout<<"Enter value of r:";
    cin>>r;
    int nFact= fact(n);
    int rFact = fact(r);
    int nrFact = fact(n-r);
    int ans = ncr(nFact, rFact, nrFact);
    return 0;
}
