//2d rrays read and print
#include<iostream>
using namespace std;
int main(){
    int rows, columns;
    cin>>rows>>columns;
    int array[rows][columns];
    for(int i=0; i<rows; i++)
    {
        for(int j=0; j<columns; j++)
        {
            cin>>array[i][j];
        }
    }
    for(int i=0; i<rows; i++)
    {
        for(int j=0; j<columns; j++)
        {
            cout<<array[i][j]<<" ";
        }cout<<endl;
    }
    return 0;
}
