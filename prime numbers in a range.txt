//prime numbers in range
#include<iostream>
#include<cmath>
using namespace std;

bool isPrime(int n)
{
    int i;
    if(n%i==0)
    {
        return false;
        
    }
    return true;
   
    
    
}

int main()
{
    int i, a, b;
    cout<<"Enter the range: ";
    cin>>a>>b;
    for(i=a; i<=b; i++)
    {
        if(isPrime(i))
        {
            cout<<i<<endl;
            
        }
    }
    return 0;
}
