#include<iostream>
using namespace std;
int main()
{
    int i, j, n=4;
    for(i=1; i<=n; i++)
    {
        for(j=1; j<=i; j++)
        {
            cout<<"*";
        }
        for(j=i+1; j<=(2*n)-(i+1); j++)
        {
            cout<<" ";
        }
        for(j=(2*n)-i; j<=2*n-1; j++)
        {
            cout<<"*";
        }
        cout<<endl;
        
        
    }
    for(i=n; i>=1; i--)
    {
        for(j=1; j<=i; j++)
        {
            cout<<"*";
        }
        for(j=i+1; j<=(2*n)-(i+1); j++)
        {
            cout<<" ";
        }
        for(j=(2*n)-i; j<=2*n-1; j++)
        {
            cout<<"*";
        }
        cout<<endl;
    }
    return 0;
    
    
}
