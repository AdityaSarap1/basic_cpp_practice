//lcm of 5 and 7

#include<iostream>
using namespace std;
int main()
{
    //while loop
    int i = 7;
    while(true){
        if(i%5==0){
            cout<<i<<endl;
            break;
        }
        else{
            i += 7;
        }
        
    }
    //for loop
    for(i=5;; i += 5)
    {
        if(i%7==0){
            cout<<i;
            break;
        }
        
        
    }
    return 0;
}
