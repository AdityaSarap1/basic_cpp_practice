//Fibonacci sequence//factorial
#include<iostream>
using namespace std;

int fact(int n)
{
    int initial=1;
    for(int i=2; i<=n; i++)
    {
        initial=initial*i;
        
    }
    return initial;
}

int main()
{
    cout<<"Enter your number: "<<endl;
    int n;
    cin>>n;
    int ans=fact(n);
    cout<<ans<<endl;
}

void fib(int num)
{
    int t1=0, t2=1, nextTerm;
    for(int i=1; i<=num; i++)
    {
        cout<<t1<<endl;
        nextTerm = t1+t2;
        t1=t2;
        t2=nextTerm;
        
    }
    return;
    
    
}

int main()
{
    int n;
    cout<<"Enter number of fibonacci digits: ";
    cin>>n;
    fib(n);
    return 0;
}
