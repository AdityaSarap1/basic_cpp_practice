//To learn functions
#include <iostream>
using namespace std;

int add(int num1, int num2)
{
    int sum = num1 + num2;
    return sum; 
}

int main()
{
    int a, b;
    cout<<"Enter two numbers: ";
    cin>>a>>b;
    int ans = add(a,b);
    cout<<"The sum of the two numbers is "<<ans<<endl;
    

    return 0;
}
